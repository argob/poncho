Un solo punto

Mapa de un solo punto sin sin el uso de popUp o slider. 

<!-- SCRIPTS -->
<script>
    const latitud = -34.60759;
    const longitud = -58.37393;
    const mapOptions = {
        scope: "poncho-map",
        theme_tool: true,
        no_info: true,
        map_zoom: 17,  
        map_view: [latitud, longitud]
    };
    const pm = new PonchoMap([{latitud, longitud}], mapOptions);
    pm.render();
</script>
<!-- / SCRIPTS -->
